Toggle

A button that toggles between pressed and unpressed states.

Preview

Code

<!-- Toggle (unpressed by default) -->
<div x-data="mwToggle()">
  <button class="mw-toggle"
          :data-state="pressed ? 'on' : 'off'"
          @click="toggle()"
          :aria-pressed="pressed">
    <svg ...>Bold icon</svg>
  </button>
</div>

<!-- Toggle (pressed by default) -->
<div x-data="mwToggle(true)">
  <button class="mw-toggle"
          :data-state="pressed ? 'on' : 'off'"
          @click="toggle()"
          :aria-pressed="pressed">
    <svg ...>Italic icon</svg>
  </button>
</div>

<!-- Toggle with text label -->
<div x-data="mwToggle()">
  <button class="mw-toggle !aspect-auto px-3 gap-2"
          :data-state="pressed ? 'on' : 'off'"
          @click="toggle()"
          :aria-pressed="pressed">
    <svg ...>Icon</svg>
    <span class="text-sm">Wrap</span>
  </button>
</div>
