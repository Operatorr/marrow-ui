Code

<!-- Headings -->
<h1 class="font-heading text-4xl font-extrabold tracking-tight lg:text-5xl">Heading 1</h1>
<h2 class="font-heading text-3xl font-semibold tracking-tight border-b pb-2">Heading 2</h2>
<h3 class="font-heading text-2xl font-semibold tracking-tight">Heading 3</h3>
<h4 class="font-heading text-xl font-semibold tracking-tight">Heading 4</h4>
<h5 class="font-heading text-lg font-semibold tracking-tight">Heading 5</h5>
<h6 class="font-heading text-base font-semibold tracking-tight">Heading 6</h6>

<!-- Paragraph -->
<p class="leading-7">Standard body text with comfortable line height.</p>

<!-- Lead paragraph -->
<p class="text-lg text-muted-foreground leading-relaxed">A lead paragraph for introductions.</p>

<!-- Small text -->
<p class="text-sm text-muted-foreground">Small muted text for secondary info.</p>

<!-- Inline code -->
<p>Use the <code class="px-1.5 py-0.5 bg-muted rounded text-sm font-mono">mw-btn</code> class.</p>

<!-- Blockquote -->
<blockquote class="border-l-4 border-primary pl-4 italic text-muted-foreground">
  "A meaningful quote goes here."
  <footer class="mt-2 text-sm not-italic">&mdash; Author Name</footer>
</blockquote>

<!-- Unordered list -->
<ul class="list-disc pl-6 space-y-1 leading-7">
  <li>First item</li>
  <li>Second item</li>
  <li>Third item</li>
</ul>

<!-- Ordered list -->
<ol class="list-decimal pl-6 space-y-1 leading-7">
  <li>Step one</li>
  <li>Step two</li>
  <li>Step three</li>
</ol>
